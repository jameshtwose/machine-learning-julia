    )

    [docs]def convert_timestamp(col: pd.Series, unit="ms"):
    """
    Convert date columns from unix format to human readable datetime format.
    This function is tuned to process dataframes coming from the Neurocast DataAPI

    Parameters
    ----------
    col : pd.Series
        pandas column which contains the timestamp in unix format

    unit : str (default: 'ms')
        unit of the arg (D,s,ms,us,ns) denote the unit, which is an integer or float number.
        This will be based off the origin. Example, with unit=’ms’ and origin=’unix’ (the default),
        this would calculate the number of milliseconds to the unix epoch start.

    Returns
    -------
    col : pd.Series
        pandas series with dtype=datetime

    Example
    -------
    >>> # from internal_utils import convert_timestamp
    >>> # 1572436800000 = 2019-10-30 12:00:00 | 1508360400000 = 2017-10-18 21:00:00| 599950800000 = 1989-01-04 21:00:00
    >>> df = pd.DataFrame({'numbers': ['date1', 'date2', 'date3'],
                          'unix_time': [1572436800000, 1508360400000, 599950800000]}
                         )
    >>> df['Timestamp'] = convert_timestamp(df['unix_time'])

    """

    col = pd.to_datetime(col, origin="unix", unit=unit, infer_datetime_format=True)

    # Check NaT
    if (col == pd.NaT).any():
        warnings.warn("WARNING: NaT were detected", stacklevel=2)

    return col
    



    [docs]def add_missings_or_negatives(
    variable: pd.Series, percentage: float, method: str, random_seed: int = None
):
    r"""Add missing values or negatives at random indices of a series

    Parameters
    ----------
    variable: pandas.Series
        series with the data

    percentage: float
        percentage of indices that should be transformed to a negative or missing

    method: str
        type of transformation, options: {'missings','negatives'}

    random_seed: int
        optional, to set a seed for the random selection of indices

    Returns
    ----------
    pandas.Series
        transformed series

    Examples
    ----------
    >>> data = sns.load_dataset(name = "mpg")
    >>> data['cylinders_with_missings'] = add_missings_or_negatives(data['cylinders'], percentage=0.02, method='missings', random_seed=5)

    """
    # Sample a random selection of indices
    random.seed(random_seed)
    random_selection = random.sample(
        range(len(variable)), round(percentage * len(variable))
    )

    # Copy the original data
    variable = variable.copy()

    if method not in ["missings", "negatives"]:
        raise ValueError("type not recognized. Use 'missings' or 'negatives'")

    if method == "missings":
        # Add missing values at random indices
        variable.loc[random_selection] = np.repeat(np.nan, len(random_selection))

    if method == "negatives":
        # Add negative values at random indices
        variable.loc[random_selection] = variable.loc[random_selection] * -1

    return variable
    



    [docs]def log_step(func):
    """
    Wrap (decorator) function with logger which adds shape of pd.DataFrame and execution time of function.

    Add @log_step before function in the following way:

    @log_step
    def any_function():
        ...

    Parameters
    ----------
    func: python function

    Returns
    -------
    output of function and additional logging info

    """

    @wraps(func)
    def wrapper(*args, **kwargs):
        tic = dt.datetime.now()
        result = func(*args, **kwargs)
        time_taken = str(dt.datetime.now() - tic)
        logger.info(f"[{func.__name__}] shape={result.shape} time={time_taken}")
        return result

    return wrapper
    



    [docs]def log_step_aws(func):
    """
    Wrap (decorator) function with logger which adds end- and start date, time range and number of selected users
    of the extracted aws data.
    Add @log_step_aws before aws function in the following way:
    @log_step_aws
    def aws_data_extraction():
        ...
    Parameters
    ----------
    func: python function
    Returns
    -------
    output of function and additional logging info
    """

    @wraps(func)
    def wrapper(*args, **kwargs):
        result = func(*args, **kwargs)
        if ("Timestamp" in result.columns) and ("user_id" in result.columns):
            start_date = convert_timestamp(result.Timestamp.min())
            end_date = convert_timestamp(result.Timestamp.max())
            n_users = result.user_id.nunique()
            logger.info(
                f"""df_spec: start_date={start_date} end_date={end_date} time_range={(end_date - start_date)}
                        n_selected_users= {n_users}"""
            )
        return result

    return wrapper
    



    [docs]def save_figure(fig, name=None, dir=None, quality=95, dpi=600, **kwargs):
    r"""Save figure with a certain name to a directory. Basically a wrapper for `matplotlib.pyplt.savefig`

    Parameters
    ----------
    fig: matplotlib.figure.Figure
    name: str or None
    dir: str or None
        Directory where to save the figure. If None, saved to current working directory
        Can be a relative path or an absolute path
    quality: int
        Image quality, ranging from 1 till 95
    dpi: int or None or 'figure' (default: 600)
        The resolution in dots per inch. If None, defaults to rcParams["savefig.dpi"] = 'figure'.
        If 'figure', uses the figure's dpi value.

    Examples
    ----------
    >>> fig = plt.figure()
    >>> save_figure(fig, name='t-SNE.jpg', dir='/home/user/Downloads')
    """

    if not name:
        warnings.warn(
            "Figure did not have a name, saved using default name'figure'", stacklevel=2
        )
        name = "figure"

    if not dir:
        dir = ""

    _ = fig.savefig(fname=dir + name, quality=quality, dpi=dpi, **kwargs)

    r"""Function to print things in method chaining, leaving the dataframe untouched

    Parameters
    ----------
    data : pandas.DataFrame
    fn: lambda
        function that takes a pandas.DataFrame and that creates output to be printed
    msg: str or None

    ----------
    >>> from neuropy.utils import chain_snap
    >>> import pandas as pd

    >>> df = df.pipe(chain_snap, msg='Shape of the dataframe:')
    >>> df = df.pipe(chain_snap, fn = lambda df: df['letter'].value_counts(), msg="Frequency of letters:")
    >>> df = df.pipe(chain_snap, fn = lambda df: df.loc[df['letter']=='c'], msg="Dataframe where letter is c:")
    """

    if msg:
        print("\n" + msg)
    print(fn(data))
    return data
    



    [docs]def drop_with_regex(data, regex: str, axis=1):
    r"""Drop columns or rows from dataframe that match regex (as implemented in pd.DataFrame.filter )

    Parameters
    ----------
    data: pandas.DataFrame
    regex: str (regular expression)
        Drop labels from axis for which re.search(regex, label) == True.
    axis: {0 or 'index', 1 or 'columns'}, default: 1
        Whether to drop labels from the index (0 or 'index') or columns (1 or 'columns').
    """
    if axis in [1, "columns"]:
        labels = data.filter(regex=regex, axis=axis).columns
    elif axis in [0, "index"]:
        labels = data.filter(regex=regex, axis=axis).index
    else:
        raise ValueError("Axis should be one of {0: index, 1: columns}")
    return data.drop(labels, axis=axis)
