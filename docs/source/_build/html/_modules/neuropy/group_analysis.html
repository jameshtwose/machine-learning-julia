   - **roc_analysis():** Compute ROC analysis [medical field version] <br>

import warnings
from typing import Union

import matplotlib.pyplot as plt
import numpy as np
import pandas as pd
from sklearn.metrics import auc

    y_benchmark: np.array,
    delta_feature: np.array,
    threshold_array: Union[np.array, None] = None,
    plot: bool = True,
    anchor_name: str = None,
    feature_name: str = None,
    figsize=(11.7, 8.27),
):
    """
    Compute ROC analysis [medical field version]
    Ref: Receiver Operating Characteristic (ROC) Curve Analysis for Medical Diagnostic Test Evaluation
    DOI: https://www.ncbi.nlm.nih.gov/pmc/articles/PMC3755824/

    ----------
    y_benchmark: np.array, pd.Series, list
        array with binary values (0, 1) relative to the benchmark/anchor
    delta_feature: np.array
        array containing the difference of a feature feature at two difference time points, e.g. between m00 and m03
    threshold_array: Union[np.array, None] (default: None)
        array contining threshold values to evaluate `delta_scores` on
    plot : bool (default: True)
        plot roc_curve
    anchor_name : str (default: None),
        add to plot the name of y_benchmark
    feature_name : str (default: None),
        add to plot the name of feature
    figsize : tuple (default: (11.7, 8.27))
        set plot size

    Returns
    ----------
    summary: pandas.DataFrame
        Summary dataframe with with columns true positive rate (tpr), false positive rate (fpr),
        true negative rate (tnr), Youden index
    roc_auc: float
        the area under the receiving operating characteristic (ROC)
    fig: matplotlib.figure.Figure (optional)
        Figure if plot == True, else None

    Notes
    -----
    Threshold implementation is equivalent to sklearn's `roc_curve()` with `drop_intermediate=False`.

    Example
    ---------
    >>> import pandas as pd
    >>> from neuropy.group_analysis import medical_roc
    >>> df = pd.DataFrame(data={'y_benchmark': [0, 0, 1, 1],
    ...                         'feature': [0.1, 0.4, 0.35, 0.8]})
    >>> summary, roc_auc, fig = medical_roc(df['y_benchmark'], df['feature'])
    >>> # plt.show()

    """
    # cast input if necessary
    if not isinstance(y_benchmark, np.ndarray):
        y_benchmark = np.array(y_benchmark)
    if not isinstance(delta_feature, np.ndarray):
        delta_feature = np.array(delta_feature)
    if threshold_array is not None and not isinstance(threshold_array, np.ndarray):
        threshold_array = np.array(threshold_array)

    # check whether input series have the same length
    if len(delta_feature) != len(y_benchmark):
        raise ValueError("Input arguments must have the same length.")

    # remove nans if any
    if np.isnan(y_benchmark).any() or np.isnan(delta_feature).any():
        warnings.warn("NaN values detected in input arrays.")
        # remove element-wise
        is_nan = np.logical_or(np.isnan(y_benchmark), np.isnan(delta_feature))
        y_benchmark, delta_feature = y_benchmark[~is_nan], delta_feature[~is_nan]

    # check if arrays are empty after removing NaNs
    if not y_benchmark.size > 0 or not delta_feature.size > 0:
        raise ValueError("Input arguments must contain valid numbers.")

    # sort input arrays according to delta scores
    sorted_idx = delta_feature.argsort()
    y_benchmark, delta_feature = y_benchmark[sorted_idx], delta_feature[sorted_idx]

    if threshold_array is None:
        # select only unique delta scores to avoid repeated error counts.
        # note: include last value to match array length
        unique_deltas = np.where(np.diff(delta_feature))[0]
        threshold_idxs = np.r_[unique_deltas, len(y_benchmark) - 1]
        threshold_array = delta_feature[threshold_idxs]

        # add an extra threshold position to make sure that the curve starts at (0, 0)
        threshold_array = np.r_[threshold_array, threshold_array[-1] + 1]

    tpr = []  # true positive rate a.k.a. sensitivity
    fpr = []  # false positive rate a.k.a. fall-out
    for threshold in threshold_array:
        threshold_bool = delta_feature < threshold
        tp = sum((threshold_bool == 1) & (y_benchmark == 1))  # true positive
        fp = sum((threshold_bool == 1) & (y_benchmark == 0))  # false positive
        fn = sum((threshold_bool == 0) & (y_benchmark == 1))  # false negative
        tn = sum((threshold_bool == 0) & (y_benchmark == 0))  # true negative

        # append the output to false positive rate list and true positive rate list
        fpr.append(tn / (tn + fp))
        tpr.append(fn / (tp + fn))

    # make dataframe
    summary_fcn = pd.DataFrame(
        data={
            "threshold": threshold_array,
            "fpr": fpr,  # fall-out
            "tpr": tpr,  # sensitivity
        }
    ).round(2)

    summary_fcn["tnr"] = 1 - summary_fcn["fpr"]  # specificity
    summary_fcn["Youden_index"] = summary_fcn["tpr"] + summary_fcn["tnr"] - 1

    if len(threshold_array) > 1:
        # compute auc
        auc_fcn = auc(summary_fcn["fpr"].values, summary_fcn["tpr"].values)
    else:
        auc_fcn = 0

    # get best Youden_index
    best_youden = summary_fcn.iloc[summary_fcn["Youden_index"].idxmax()]

    fig = None
    if plot:
        fig = plt.figure(figsize=figsize)
        lw = 3
        text_size = 15
        plt.plot(
            summary_fcn["fpr"],
            summary_fcn["tpr"],
            color="darkorange",
            marker="o",
            markerfacecolor="navy",
            markersize=8,
            lw=lw,
            label="ROC curve (area = %0.2f)" % auc_fcn,
        )

        plt.plot([0, 1], [0, 1], color="navy", lw=lw - 1, linestyle="--")

        plt.plot(
            best_youden["fpr"],
            best_youden["tpr"],
            marker="s",
            markerfacecolor="red",
            markersize=8,
            label="Best Youden index = %0.2f" % best_youden["Youden_index"],
        )

        plt.vlines(
            x=best_youden["fpr"],
            ymin=0,
            ymax=best_youden["tpr"],
            lw=1,
            colors="k",
            linestyles="dotted",
            label="MCID = %0.2f" % best_youden["threshold"],
        )

        plt.hlines(
            y=best_youden["tpr"],
            xmin=0,
            xmax=best_youden["fpr"],
            lw=1,
            colors="k",
            linestyles="dotted",
        )

        plt.xlim([0, 1.0])
        plt.ylim([0, 1.05])
        plt.xlabel("FPR (1 - Specificity)", size=text_size)
        plt.ylabel("TPR (Sensitivity)", size=text_size)
        plt.title("Receiver operating characteristic", size=text_size)
        plt.legend(loc="lower right", fontsize="large")
        plt.xticks(size=text_size)
        plt.yticks(size=text_size)

        if anchor_name is not None and feature_name is not None:
            plt.title(
                f'Receiver operating characteristic\n anchor = {anchor_name.replace("_", " ")}  |  feature = {feature_name.replace("_", " ")}',
                size=text_size,
            )
        else:
            plt.title("Receiver operating characteristic", size=text_size)

    return summary_fcn, auc_fcn, fig
