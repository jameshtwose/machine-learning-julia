    [docs]def variance_threshold(
    data: pd.DataFrame, threshold=0.001, show=False, plot=False, figsize=(11.7, 8.27)
):
    r"""Perform dimensionality reduction using variance threshold. The method considers numerical features only.
        Rows with any missing values will be discarded.
    ----------
    data: pandas.DataFrame
        Dataframe with variables in columns, cases in rows
    threshold: float (default: 0.001)
        Features with a variance lower than threshold will be flagged.
    show: bool (default: False)
        whether to show the summary
    plot: bool (default: False)
        whether to plot the box plot with normalized features
    figsize: tuple (default: (11.7, 8.27))
        Width and height of the figure in inches

    Returns
    ----------
    summary: pandas.DataFrame
        Dataframe containing a feature selection summary:
        [normalized-variance feature-type  variance-threshold  to-discard]
    boxplot: Figure
        Properties of the gigure can be changed later, e.g. boxplot.axes[0].set_ylim(0,10) to change the limits. If
        plot == False, boxplot will be None.

    Examples
    ----------
    >>> from neuropy.dimensionality_reduction import variance_threshold
    >>> data = sns.load_dataset(name = 'titanic')
    >>> summary, _ = variance_threshold(data = data, threshold = 0.05, show = False, plot = True)
    """
    # TODO address normalization
    # select only numerical feature, drop row with any nan and normalize feature such that mean(feature) = 1
    data = data.copy()
    data = data.select_dtypes(exclude="bool")
    normalized_data = (
        data.select_dtypes(include="number").dropna(axis="index", how="any")
        / data.mean()
    )

    summary = normalized_data.var().to_frame()
    summary.columns = ["normalized-variance"]
    summary["feature-type"] = data.dtypes
    summary["variance-threshold"] = threshold
    summary["to-discard"] = summary["normalized-variance"] < threshold
    summary = summary[
        ["feature-type", "normalized-variance", "variance-threshold", "to-discard"]
    ]
    _ = summary.sort_values(by="normalized-variance", inplace=True)

    if show:
        print(summary)

    boxplot = None

    if plot:
        # plot red feature < threshold & green feature >= threshold
        dict_palette = dict(
            zip(
                summary.index,
                summary["to-discard"].map({True: "tomato", False: "springgreen"}),
            )
        )

        boxplot, ax = plt.subplots(figsize=figsize)
        _ = sns.boxplot(
            ax=ax, data=normalized_data, palette=dict_palette, order=summary.index
        )
        _ = plt.setp(ax.get_xticklabels(), rotation=90)
        _ = plt.title(
            f"Feature selection via variance threshold\n original features = {normalized_data.shape[1]} "
            f"| selected features = {summary.loc[summary['to-discard'] == False].shape[0]}"
        )
        _ = plt.grid()
        _ = plt.tight_layout()
        _ = plt.ylabel("Normalized data")
        patch_discard = mpatches.Patch(color="tomato", label="feature to discard")
        patch_keep = mpatches.Patch(color="springgreen", label="selected feature")
        plt.legend(handles=[patch_keep, patch_discard])

    return summary, boxplot

    data: pd.DataFrame, threshold=5, show=False, plot=False, figsize=(25, 10)
):
    r"""Perform dimensionality reduction using missing value threshold.
    The method considers both numerical and categorical features.
    ----------
    data: pandas.DataFrame
        Dataframe with variables in columns, cases in rows
    threshold: float (default: 5)
        Features with a higher missing value percentage than threshold will be flagged
    show: bool (default: False)
        whether to show the summary
    plot: bool (default: False)
        whether to plot the heatmap with missing values
    figsize: tuple (default: (25, 10))
        Width and height of the figure in inches

    Returns
    ----------
    summary: pandas.DataFrame
        Dataframe containing a feature selection summary:
        ['feature-type', 'missing-value %', 'missing-threshold %', 'to-discard']
    missplot: Figure
        A figure is returned if plot == True, else None

    Examples
    ----------
    >>> from neuropy.dimensionality_reduction import missing_value_threshold
    >>> titanic = sns.load_dataset(name='titanic')
    >>> summary, fig = missing_value_threshold(data=titanic, threshold=5, show=False, plot=True)
    """

    summary = round(data.isna().sum() / len(data) * 100, 2)
    summary = summary.to_frame()
    summary.columns = ["missing-value %"]
    summary["feature-type"] = data.dtypes
    summary["missing-threshold %"] = threshold
    summary["to-discard"] = summary["missing-value %"] > threshold
    summary = summary[
        ["feature-type", "missing-value %", "missing-threshold %", "to-discard"]
    ]
    _ = summary.sort_values(by="missing-value %", ascending=False, inplace=True)

    if show:
        print(summary)

    missplot = None

    if plot:
        missplot = msno_matrix(df=data, figsize=figsize)

    return summary, missplot
    



    [docs]def nullity_sort(df, sort=None, axis="columns"):
    """
    COPIED FROM MISSINGNO
    Sorts a DataFrame according to its nullity, in either ascending or descending order.
    :param df: The DataFrame object being sorted.
    :param sort: The sorting method: either "ascending", "descending", or None (default).
    :return: The nullity-sorted DataFrame.
    """
    if sort is None:
        return df
    elif sort not in ["ascending", "descending"]:
        raise ValueError(
            'The "sort" parameter must be set to "ascending" or "descending".'
        )

    if axis not in ["rows", "columns"]:
        raise ValueError('The "axis" parameter must be set to "rows" or "columns".')

    if axis == "columns":
        if sort == "ascending":
            return df.iloc[np.argsort(df.count(axis="columns").values), :]
        elif sort == "descending":
            return df.iloc[np.flipud(np.argsort(df.count(axis="columns").values)), :]
    elif axis == "rows":
        if sort == "ascending":
            return df.iloc[:, np.argsort(df.count(axis="rows").values)]
        elif sort == "descending":
            return df.iloc[:, np.flipud(np.argsort(df.count(axis="rows").values))]
    



    [docs]def nullity_filter(df, filter=None, p=0, n=0):
    """
    COPIED FROM MISSINGNO
    Filters a DataFrame according to its nullity, using some combination of 'top' and 'bottom' numerical and
    percentage values. Percentages and numerical thresholds can be specified simultaneously: for example,
    to get a DataFrame with columns of at least 75% completeness but with no more than 5 columns, use
    `nullity_filter(df, filter='top', p=.75, n=5)`.
    :param df: The DataFrame whose columns are being filtered.
    :param filter: The orientation of the filter being applied to the DataFrame. One of, "top", "bottom",
    or None (default). The filter will simply return the DataFrame if you leave the filter argument unspecified or
    as None.
    :param p: A completeness ratio cut-off. If non-zero the filter will limit the DataFrame to columns with at least p
    completeness. Input should be in the range [0, 1].
    :param n: A numerical cut-off. If non-zero no more than this number of columns will be returned.
    :return: The nullity-filtered `DataFrame`.
    """
    if filter == "top":
        if p:
            df = df.iloc[:, [c >= p for c in df.count(axis="rows").values / len(df)]]
        if n:
            df = df.iloc[:, np.sort(np.argsort(df.count(axis="rows").values)[-n:])]
    elif filter == "bottom":
        if p:
            df = df.iloc[:, [c <= p for c in df.count(axis="rows").values / len(df)]]
        if n:
            df = df.iloc[:, np.sort(np.argsort(df.count(axis="rows").values)[:n])]
    return df

    """
    A matrix visualization of the nullity of the given DataFrame. COPIED from missingo package
    :param df: The `DataFrame` being mapped.
    :param filter: The filter to apply to the heatmap. Should be one of "top", "bottom", or None (default).
    :param n: The max number of columns to include in the filtered DataFrame.
    :param p: The max percentage fill of the columns in the filtered DataFrame.
    :param sort: The row sort order to apply. Can be "ascending", "descending", or None.
    :param figsize: The size of the figure to display.
    :param fontsize: The figure's font size. Default to 16.
    :param labels: Whether or not to display the column names. Defaults to the underlying data labels when there are
    50 columns or less, and no labels when there are more than 50 columns.
    :param sparkline: Whether or not to display the sparkline. Defaults to True.
    :param width_ratios: The ratio of the width of the matrix to the width of the sparkline. Defaults to `(15, 1)`.
    Does nothing if `sparkline=False`.
    :param color: The color of the filled columns. Default is `(0.25, 0.25, 0.25)`.
    :return: If `inline` is False, the underlying `matplotlib.figure` object. Else, nothing.
    """
    df = nullity_filter(df, filter=filter, n=n, p=p)
    df = nullity_sort(df, sort=sort, axis="columns")

    height = df.shape[0]
    width = df.shape[1]

    # z is the color-mask array, g is a NxNx3 matrix. Apply the z color-mask to set the RGB of each pixel.
    z = df.notnull().values
    g = np.zeros((height, width, 3))

    g[z < 0.5] = [1, 1, 1]
    g[z > 0.5] = color

    # Set up the matplotlib grid layout. A unary subplot if no sparkline, a left-right splot if yes sparkline.
    if ax is None:
        plt.figure(figsize=figsize)
        if sparkline:
            gs = gridspec.GridSpec(1, 2, width_ratios=width_ratios)
            gs.update(wspace=0.08)
            ax1 = plt.subplot(gs[1])
        else:
            gs = gridspec.GridSpec(1, 1)
        ax0 = plt.subplot(gs[0])
    else:
        if sparkline is not False:
            warnings.warn(
                "Plotting a sparkline on an existing axis is not currently supported. "
                "To remove this warning, set sparkline=False."
            )
            sparkline = False
        ax0 = ax

    # Create the nullity plot.
    ax0.imshow(g, interpolation="none")

    # Remove extraneous default visual elements.
    ax0.set_aspect("auto")
    ax0.grid(b=False)
    ax0.xaxis.tick_top()
    ax0.xaxis.set_ticks_position("none")
    ax0.yaxis.set_ticks_position("none")
    ax0.spines["top"].set_visible(False)
    ax0.spines["right"].set_visible(False)
    ax0.spines["bottom"].set_visible(False)
    ax0.spines["left"].set_visible(False)

    # Set up and rotate the column ticks. The labels argument is set to None by default. If the user specifies it in

    if labels or (labels is None and len(df.columns) <= 50):
        ha = "left"
        ax0.set_xticks(list(range(0, width)))
        ax0.set_xticklabels(list(df.columns), rotation=45, ha=ha, fontsize=fontsize)
    else:
        ax0.set_xticks([])

    # Adds Timestamps ticks if freq is not None, else set up the two top-bottom row ticks.
    if freq:

                df.index.to_timestamp().date[0],
                df.index.to_timestamp().date[-1],
                freq=freq,
            ).values

                df.index.to_timestamp().date[0],
                df.index.to_timestamp().date[-1],
                freq=freq,
            ).map(lambda t: t.strftime("%Y-%m-%d"))

        elif type(df.index) == pd.DatetimeIndex:
            ts_array = pd.date_range(
                df.index.date[0], df.index.date[-1], freq=freq
            ).values

            ts_ticks = pd.date_range(
                df.index.date[0], df.index.date[-1], freq=freq
            ).map(lambda t: t.strftime("%Y-%m-%d"))
        else:
            raise KeyError("Dataframe index must be PeriodIndex or DatetimeIndex.")
        try:
            for value in ts_array:
                ts_list.append(df.index.get_loc(value))
        except KeyError:
            raise KeyError("Could not divide time index into desired frequency.")

        ax0.set_yticks(ts_list)
        ax0.set_yticklabels(ts_ticks, fontsize=int(fontsize / 16 * 20), rotation=0)
    else:
        ax0.set_yticks([0, df.shape[0] - 1])
        ax0.set_yticklabels(
            [1, df.shape[0]], fontsize=int(fontsize / 16 * 20), rotation=0
        )

    # Create the inter-column vertical grid.
    in_between_point = [x + 0.5 for x in range(0, width - 1)]
    for in_between_point in in_between_point:
        ax0.axvline(in_between_point, linestyle="-", color="white")

    if sparkline:
        # Calculate row-wise completeness for the sparkline.
        completeness_srs = df.notnull().astype(bool).sum(axis=1)
        x_domain = list(range(0, height))
        y_range = list(reversed(completeness_srs.values))
        min_completeness = min(y_range)
        max_completeness = max(y_range)
        min_completeness_index = y_range.index(min_completeness)
        max_completeness_index = y_range.index(max_completeness)

        # Set up the sparkline, remove the border element.
        ax1.grid(b=False)
        ax1.set_aspect("auto")
        # GH 25
        ax1.set_facecolor((1, 1, 1))
        ax1.spines["top"].set_visible(False)
        ax1.spines["right"].set_visible(False)
        ax1.spines["bottom"].set_visible(False)
        ax1.spines["left"].set_visible(False)
        ax1.set_ymargin(0)

        # Plot sparkline---plot is sideways so the x and y axis are reversed.
        ax1.plot(y_range, x_domain, color=color)

        if labels:
            # Figure out what case to display the label in: mixed, upper, lower.
            label = "Data Completeness"
            if str(df.columns[0]).islower():
                label = label.lower()
            if str(df.columns[0]).isupper():
                label = label.upper()

            # Set up and rotate the sparkline label.

            ax1.set_xticks(
                [min_completeness + (max_completeness - min_completeness) / 2]

            ax1.set_xticklabels([label], rotation=45, ha=ha, fontsize=fontsize)
            ax1.xaxis.tick_top()
            ax1.set_yticks([])
        else:
            ax1.set_xticks([])
            ax1.set_yticks([])

        # Add maximum and minimum labels, circles.
        ax1.annotate(
            max_completeness,
            xy=(max_completeness, max_completeness_index),
            xytext=(max_completeness + 2, max_completeness_index),
            fontsize=int(fontsize / 16 * 14),
            va="center",
            ha="left",
        )
        ax1.annotate(
            min_completeness,
            xy=(min_completeness, min_completeness_index),
            xytext=(min_completeness - 2, min_completeness_index),
            fontsize=int(fontsize / 16 * 14),
            va="center",
            ha="right",
        )

        ax1.set_xlim(
            [min_completeness - 2, max_completeness + 2]
        )  # Otherwise the circles are cut off.
        ax1.plot(
            [min_completeness],
            [min_completeness_index],
            ".",
            color=color,
            markersize=10.0,
        )
        ax1.plot(
            [max_completeness],
            [max_completeness_index],
            ".",
            color=color,
            markersize=10.0,
        )

        # Remove tick mark (only works after plotting).
        ax1.xaxis.set_ticks_position("none")

    if inline:
        warnings.warn(
            "The 'inline' argument has been deprecated, and will be removed in a future version "
            "of missingno."
        )
    else:
        return ax0
    



    [docs]def chisquare_threshold(
    X: pd.DataFrame,
    y: pd.Series,
    threshold=0.05,
    show=False,
    plot=False,
    figsize=(11.7, 8.27),
):
    r"""Perform dimensionality reduction using chisquare. The method considers is applied only to categorical features.
        the target variable y need to either either categical or integer type. Rows with any missing values will be discarded.
    ----------
    data: pandas.DataFrame
        Dataframe with variables in columns, cases in rows
    threshold: float (default: 0.05)
        p-value threshold which is used to assess which feature to discard
    show: bool (default: False)
        whether to show the summary
    plot: bool (default: False)
        whether to plot the bar plot with normalized features
    figsize: tuple (default: (11.7, 8.27))
        Width and height of the figure in inches

    Returns
    ----------
    summary: pandas.DataFrame
        Dataframe containing a feature selection summary:
        [normalized-variance feature-type  variance-threshold  to-discard]
    barplot: Figure
        A figure is returned if plot == True, else None


    Examples
    ----------
    >>> data = sns.load_dataset(name = 'titanic')
    >>> X = data.drop(labels=['survived'], axis='columns')
    >>> y = data['survived']
    >>> summary, fig = chisquare_threshold(X = X, y = y, show = True, plot = True)

    """

    # start function
    X = X.copy()
    y = y.copy()

    # discard all numerical features and keep track of the original dtypes
    X = X.select_dtypes(exclude="float")
    dtype_X = X.dtypes

    # check consistecy w.r.t. dtypes and samples
    if len(X) != len(y):
        raise RuntimeError("Lengths of X and y differ")
    if not (
        pd.api.types.is_categorical_dtype(y)
        or pd.api.types.is_integer_dtype(y)
        or pd.api.types.is_string_dtype(y)
    ):
        raise RuntimeError(
            'y need to be either type(y) = "category" | "int". Use y.astype(int) '
        )

    # Missing values handling in y
    if y.isna().any():
        # Remove all observations that have missing y values
        warnings.warn(
            "WARNING: y included missing values. These observations are dropped in both y and X",
            stacklevel=2,
        )
        correct_y_indices = pd.notna(y)
        y = y[correct_y_indices]
        X = X[correct_y_indices]

    # Drop observations with missing values in features
    if X.isna().any(axis=None):
        warnings.warn(
            "WARNING: X included missing values. These observations are "
            "dropped in both y and X",
            stacklevel=2,
        )
        correct_X_indices = pd.notna(X).all(axis=1)
        y = y[correct_X_indices]
        X = X[correct_X_indices]

    # convert any 'object' column to 'category'. Boolean don't need to be converted
    X = pd.concat(
        [
            X.select_dtypes(exclude=["object"]),
            X.select_dtypes(include=["object"]).apply(
                pd.Series.astype, dtype="category"
            ),
        ],
        axis=1,
    )

    # convert categories to int (frequency count), whereas include boolean without convert them
    X = pd.concat(
        [
            X.select_dtypes(exclude="category"),
            X.select_dtypes(include="category").apply(lambda x: x.cat.codes, axis=0),
        ],
        axis=1,
    )

    # compute chi2 for each feature regarding the predictive capability of y
    (chi2_value, p_value) = chi2(X, y)

    # build summary
    summary = pd.DataFrame(
        data={
            "feature-type": dtype_X,
            "chi2": chi2_value,
            "p-value": p_value,
            "stat-sign": (p_value < threshold),
            "to-discard": (p_value > threshold),
        },  # discard features based on p-value
        index=X.columns,
    )
    summary = summary.round(decimals={"chi2": 2, "p-value": 3})
    _ = summary.sort_values(by=["chi2", "p-value"], ascending=True, inplace=True)

    if show:
        print(summary)

    barplot = None

    if plot:
        # plot red feature < threshold & green feature >= threshold
        summary["color"] = "springgreen"
        summary.loc[summary["to-discard"] == True, "color"] = "tomato"
        df_palette = summary.filter(items=["color"], axis="columns")
        # create dictionary from parallel lists
        feature_name = list(df_palette.index)
        feature_color = list(df_palette["color"].values)
        dict_palette = dict(zip(feature_name, feature_color))
        # remove 'Color' column from summary
        _ = summary.drop(labels=["color"], inplace=True, axis="columns")

        # make barplot
        barplot, ax = plt.subplots(figsize=figsize)
        _ = sns.barplot(ax=ax, x=summary.index, y=summary["chi2"], palette=dict_palette)
        _ = plt.setp(ax.get_xticklabels(), rotation=90)
        _ = plt.title(
            f"Feature selection via chi2square\n original features = {X.shape[1]} "
            f"| selected features = {summary.loc[summary['to-discard'] == False].shape[0]}"
        )

        _ = ax.set(xlabel="Categorical Features", ylabel=r"$\chi^{2}$")
        _ = sns.set_context("notebook", font_scale=2.0)
        _ = plt.grid()
        _ = plt.tight_layout()

    return summary, barplot
    



    [docs]def high_correlation_threshold(
    data: pd.DataFrame,
    target: Union[str, List[str]],
    threshold=0.8,
    check_norm=False,
    method: str = "pearson",
):
    """Perform dimensionality reduction using correlation. The method will consider highly correlated features and will
        add features with a lower correlation with the target variable y to a discard list. This method uses the
        `correlation_analysis` function from neuropy.correlation.

    Parameters
    ----------
    data: pd.DataFrame
        Includes the features.
    target: str or List[str]
        Target variable, in case of multioutput problem insert a list of targets.
    threshold: float (default: 0.8)
        The threshold value for the r-value, only r-values exceeding the threshold will be considered.
    check_norm: bool (default: False)
        If True, normality will be checked for columns in `data` using `normal_check`. This influences the used method
        for correlations, i.e. Pearson or Spearman. Note: normality check ignores missing values.
    method: {'pearson', 'kendall', 'spearman'}, default 'pearson'
        Type of correlation, either Pearson's r, Spearman's rho, or Kendall's tau, implemented via respectively
        `scipy.stats.pearsonr`, `scipy.stats.spearmanr`, and `scipy.stats.kendalltau`. Ignored if check_norm is True.

    Returns
    -------
    df_summary: pd.Dataframe
        Summary dataframe with correlation analysis results and dimensionality reduction results. The correlation with
        the target(s) is/are absolute and thus do not denote directionality.
    to_discard_corr_list: List
        List of all features to discard. This list can be passed as an argument to:
            e.g. df.drop(columns=to_discard_corr_list)

    Examples
    ________
    >>> from neuropy.dimensionality_reduction import high_correlation_threshold
    >>> from sklearn import datasets
    >>> import pandas as pd
    >>> boston_data = datasets.load_boston()
    >>> df_boston = pd.DataFrame(boston_data.data,columns=boston_data.feature_names)
    >>> df_boston['target'] = pd.Series(boston_data.target)
    >>> df_summary, to_discard_corr_list = high_correlation_threshold(data=df_boston, target='target')
    """
    to_discard_corr_list = []
    to_keep_list = []

    if not isinstance(target, List):
        target = [target]

    # Runs Neuropy's correlation analysis with permutation is false.
    results, _ = correlation_analysis(
        data.drop(columns=target),
        check_norm=check_norm,
        method=method,
        permutation_test=False,
        plot_permutation=False,
    )

    # df_summary is build up on the results dict from correlation analysis.
    df_summary = (
        results["summary"]
        .sort_values(by=["r-value"], ascending=False)
        .loc[lambda d: (np.abs(d["r-value"]) > threshold)]
        .reset_index(drop=True)
    )

    for i, row in df_summary.iterrows():
        # Calculating the absolute correlation with the target (without significance).
        corr_feature_1 = (
            data[[f"{row['feature1']}"] + target]
            .corr(method=method)
            .abs()
            .loc[:, f"{row['feature1']}"][target]
            .mean()
        )
        corr_feature_2 = (
            data[[f"{row['feature2']}"] + target]
            .corr(method=method)
            .abs()
            .loc[:, f"{row['feature2']}"][target]
            .mean()
        )

        if corr_feature_1 > corr_feature_2:
            to_discard_corr_list.append(f"{row['feature2']}")
            to_keep_list.append(f"{row['feature1']}")
        else:
            to_discard_corr_list.append(f"{row['feature1']}")
            to_keep_list.append(f"{row['feature2']}")

        if len(target) == 1:
            df_summary.loc[i, "corr(feature1, target)"] = corr_feature_1
            df_summary.loc[i, "corr(feature2, target)"] = corr_feature_2
        else:
            df_summary.loc[i, "mean(corr(feature1, target))"] = corr_feature_1
            df_summary.loc[i, "mean(corr(feature2, target))"] = corr_feature_2

    df_summary["selected_feature"] = to_keep_list

    return df_summary, to_discard_corr_list
