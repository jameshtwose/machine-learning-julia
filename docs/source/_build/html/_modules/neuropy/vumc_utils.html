- **fetch_poms_dataset():** Internal function to load data quickly <br>

- **replace_default_values_with_nan():** Pipe function to remove default values <br>

def fetch_poms_dataset(
    poms_filepath: str = "20200303_APPSMS_export_DS_Friendly.csv",
) -> pd.DataFrame:
    """Fetch POMS and VUmc code tables from S3 bucket, and merge them.

    Parameters
    ----------
    poms_filepath: str (default: "20200303_APPSMS_export_DS_Friendly.csv")
        filepath to POMS table

    Returns
    -------
    pd.DataFrame
        POMS table with user ids

    Examples
    --------
    >>> df_poms = fetch_poms_dataset()

    """

    # load user id dataset
    df_user = (
        aws_data_extraction(
            filepath="VUmc/vumc_code_neurokeys_code_user_id.csv",
            index_col=0,
        )
        .rename({"Neurokeys ID": "user_id", "VUmc code": "VUmc_code"}, axis=1)
        .drop(columns=["Neurokeys code"], axis=1)
    )

    # load POMS dataset
    df_poms = aws_data_extraction(
        filepath=poms_filepath, bucket="neurokeys-vumc-ka-hoo-private"
    )

    # merge them both based on project code
    return df_poms.merge(df_user, on="VUmc_code", how="inner")
    



    [docs]def melt_multiple_test_dates(
    df_quest: pd.DataFrame,
    end_points: List,
    checkpoints: Dict[str, float] = {
        "m00": 0.0,
        "m002": 0.5,
        "m03": 3.0,
        "m06": 6.0,
        "m09": 9.0,
        "m12": 12.0,
    },
):
    """Melt dataframe columns w.r.t checkpoints (m00, m0002, m03, m06, m09, m12).
    So instead of:
            m00_CIS_total | m0002_CIS_total | m03_CIS_total | m06_CIS_total | m09_CIS_total | m012_CIS_total
                 7                8                 6               10              7               7
        Now we'll have:
            checkpoint | CIS_total
               0.0          7
               0.2          8
               3.0          6
               6.0          10
               9.0          7
               12.0         7

    Special logic for endpoints that have a m002 measurement. These endpoints consider "Questionnaires_completiondate"
    as test date. Endpoints without a m002 measurement consider 'visitdate' as test date.

    Parameters
    ----------
    df_quest: pd.DataFrame
        Dataframe that contains the end point results.
    end_point: str
        Column that specifies the scores of the end point of interest.
    checkpoints: Dict[str, float] (default: {'m00': 0., 'm002': 0.5, 'm03': 3., 'm06': 6., 'm09': 9., 'm12': 12.})
        mapping of checkpoint prefixes to floats.

    Examples
    ________
    >>> df_quest = (
    ...     fetch_poms_dataset()
    ...     .pipe(melt_multiple_test_dates, end_points=['HADS_anxiety', 'HADS_total'])
    ...     )

    Returns
    _______
    pd.DataFrame

    """

    columns_quest_date = (
        df_quest.filter(regex="m002_").columns.str.replace("m002_", "").tolist()
    )

    dfs = []
    for prefix, code in checkpoints.items():

        str_include_cols = f"|{prefix}_".join(end_points)

        df_checkpoint = (
            df_quest
            # uses replace function to remove default values in the date column.
            .pipe(replace_default_values_with_nan)
            .filter(
                regex=f"user_id|(^{prefix}_.*date$)|^age$|gender|diagnosis_ms|^{prefix}_"
                + str_include_cols
            )
            .reset_index(drop=True)
            .dropna()
            .assign(checkpoint=lambda d: code)
        )

        df_checkpoint.columns = df_checkpoint.columns.str.replace(f"{prefix}_", "")

        # Check if all end_points consider "Questionnaires_completiondate" as test date
        if all(end_point in columns_quest_date for end_point in end_points):
            df_checkpoint = df_checkpoint.assign(
                test_date=lambda d: d["Questionnaires_completiondate"]
            )
            logging.info(
                "The test date originates from the questionnaires_completiondate column"
            )

        # Check if all end_points consider "visit_date" as test date
        elif all(end_point not in columns_quest_date for end_point in end_points):
            try:
                df_checkpoint = df_checkpoint.assign(test_date=lambda d: d["visitdate"])
                logging.info("The test date originates from the visitdate column")
            except KeyError:
                pass

        else:
            raise ValueError(
                "The endpoint list can only contain endpoints that either share the visidate or the questionnaires completiondate column."
            )

        for end_point in end_points:
            if end_point in df_checkpoint.columns:
                dfs.append(df_checkpoint)

    df_melted = (
        pd.concat(dfs, ignore_index=True, sort=True, join="outer")
        .drop(columns=["Questionnaires_completiondate", "visitdate"])
        .assign(test_date=lambda d: pd.to_datetime(d["test_date"], format="%d-%m-%Y"))
    )

    return df_melted
