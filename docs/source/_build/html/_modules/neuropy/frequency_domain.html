r"""Submodule frequency_domain.py includes the following functions: <br>
   - **frequency_analysis():** Compute Fast Fourier Transform on a time series <br>
"""
import matplotlib.pyplot as plt
import numpy as np
import pandas as pd
from scipy.fftpack import fft
from scipy.fftpack import fftfreq



    [docs]def frequency_analysis(df: pd.DataFrame, fs: int, plot=True, **kwargs) -> pd.DataFrame:
    r"""Compute Fast Fourier Transform on a time series

    Parameters
    ----------
    data: pandas.DataFrame
        Dataframe which contains some numerical feature
    fs: int
        sample frequency of the signal in Hertz [Hz].
    plot: bool, optional (default: True)
        plot fourier analysis
    kwargs:
        Additional keyword arguments for plot setting matplolib style. Available setting:
         {"title": str, "xlim": tuple, "ylim": tuple, "grid": bool}.

    Returns
    ----------
    data_fourier: pandas.DataFrame
        Dataframe containing the Fourier transformed signals.
    fig: matplotlib.pyplot.Figure
        Figure is returned if plot == True, else None.

    Examples
    ----------
    >>> from neuropy.frequency_domain import frequency_analysis
    >>> # fs: sample rate | f: signal frequency | Toss: observation time
    >>> fs, f, Toss = 100, 1, 4
    >>> t = np.linspace(0, Toss, Toss * fs, endpoint=False)
    >>> # generate a signal with 2 harmonics
    >>> x_sine = np.sin(2*np.pi*f*t) + 0.5*np.cos(2*np.pi*(10*f)*t)
    >>> data_time = pd.DataFrame(data={'x_sine': x_sine},
    ...                          index=t)
    >>> plot_kwargs={"figsize":(6,6),
    ...              "title":"Fourier Analysis",
    ...              "ylim":(0,0.6),
    ...              "xlim":(0,16),
    ...              "grid":True}
    >>> data_fourier, fig = frequency_analysis(df=data_time,
    ...                                        fs=fs,
    ...                                        plot=True,
    ...                                        **plot_kwargs)
    """

    df = df.copy()

    if isinstance(df, pd.Series):
        # convert Series to DataFrame
        df = df.to_frame()
    elif isinstance(df, pd.DataFrame):
        pass
    else:
        raise TypeError("provided df can only be a pd.DataFrame or pd.Series")

    for column_name in df:
        if df[column_name].isna().sum() != 0:
            print(
                f"INFO: {df[column_name].isna().sum()} ({round(df[column_name].isna().sum() / len(df[column_name])*100, 3)}%) Nan's detected in {column_name}: values were linearly interpolated prior Fourier analysis"
            )
            _ = df[column_name].interpolate(method="linear", inplace=True)

    # create time array
    N = df.shape[0]  # get number of sample
    t = np.linspace(
        0, (1 / fs) * N, N, endpoint=False
    )  # Note: Toss=(1/fs)*N, N=Toss*fs
    df.index, df.index.name = t, "time"

    # compute fft for each columns
    data_fourier = pd.DataFrame()
    for (column_name, column_data) in df.iteritems():
        # compute fft for each column
        data_fourier[column_name + "_fft"] = fft(column_data.values)

    # get modulus and scale w.r.t number of sample (element-wise operation)
    data_fourier = (1 / N) * np.abs(data_fourier)

    # add fequency scale as index
    data_fourier.index = fftfreq(data_fourier.shape[0]) * fs
    data_fourier.index.name = "frequency"

    fig = None
    if plot:
        # set figure dimension based on kwargs
        fig = (
            plt.figure(figsize=kwargs["figsize"])
            if "figsize" in kwargs.keys()
            else plt.figure(figsize=(11.7, 8.27))
        )

        for (column_name, column_data) in data_fourier.iteritems():
            markerline, stemlines, baseline = plt.stem(
                data_fourier.index,
                column_data,
                markerfmt="o",
                label=column_name,
                use_line_collection=True,
            )
            _ = plt.setp(stemlines, "color", plt.getp(markerline, "color"))
            _ = plt.setp(stemlines, "linestyle", "solid")

        # set title, xlim, ylim as function of kwargs
        _ = (
            plt.title(kwargs["title"])
            if "title" in kwargs.keys()
            else plt.title("Time series analysis in frequency domain")
        )
        _ = plt.ylim(kwargs["ylim"]) if "ylim" in kwargs.keys() else -1
        _ = (
            plt.xlim(kwargs["xlim"])
            if "xlim" in kwargs.keys()
            else plt.xlim((0, fs / 2))
        )
        _ = plt.grid(kwargs["grid"]) if "grid" in kwargs.keys() else plt.grid()
        _ = plt.ylabel("1/N*|Y(f)|")
        _ = plt.xlabel("frequency [Hz]")
        _ = plt.legend()
        # _ = plt.show() # outcommented due to it plotting during running the pytest

    return data_fourier, fig
