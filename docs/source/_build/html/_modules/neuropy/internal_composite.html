    [docs]def construct_composite_score(
    df: pd.DataFrame,
    composite_score_dict: Dict[str, list],
    drop_input_features: bool = False,
    mask_lower_cutoff: Union[Dict[str, Union[int, float]], None] = None,
    method: Callable = np.mean,
):
    """Construct composite score with flexible type of aggregation.
    Parameters
    ----------
    df: pd.DataFrame
        dataframe containing Neurocast features
    composite_score_dict: dict
        dictionary that specifies the features for each composite score. The keys represent the composite score name,
        whereas the values are lists containing the features set used to construct the composite score.
    drop_input_features: bool (default: False)
        if True, the features used to construct the composite score are removed from the dataframe before it is
        returned.
    mask_lower_cutoff: Union[dict[str, list], None] (default: None)
        Mask elements based on a cutoff value. The pair {key:value} denotes the column name used for the condition and
        the cutoff value, respectively. For instance, {'event_count': 50} means that within the feature set used to
        construct the composite score, all the rows where event_count <= 50 will be set as np.nan.
    method: Callable (default: np.mean)
        Method passed to pd.DataFrame.apply(axis=1), used to aggregate the feature set contained in the values
        of composite_score_dict. The input can be strings (e.g. 'mean'), functions (e.g. np.mean) and
        lambda functions (e.g. lambda row: row.sum()+1)
    Returns
    -------
    pd.Dataframe
        The copy of the original dataframe containing the composite scores.
    Examples
    --------
    >>> import numpy as np
    >>> import pandas as pd
    >>> from neuropy.internal_composite import construct_composite_score
    >>> import seaborn as sns
    >>> # load dataset
    >>> np.random.seed(seed=42)
    >>> df = (sns
    ...        .load_dataset(name='iris')
    ...        .assign(event_count = np.random.rand(150))
    ...       )
    >>> # construct two composited scores named SEPAL and PETAL using mean aggregation,
    >>> # and mask all value where event_count <= 0.2
    >>> df_out = construct_composite_score(df=df,
    ...                                   composite_score_dict={'SEPAL': ['sepal_length','sepal_width'],
    ...                                                         'PETAL': ['petal_length','petal_width']},
    ...                                   mask_lower_cutoff={'event_count': 0.2},
    ...                                   method=np.mean)
    >>> # construct two composite scores named SEPAL and PETAL using np.linalg.norm(ord=2).
    >>> df_out = construct_composite_score(df=df,
    ...                                   composite_score_dict={'SEPAL': ['sepal_length','sepal_width'],
    ...                                                         'PETAL': ['petal_length','petal_width']},
    ...                                   method=lambda x: np.linalg.norm(x=x, ord=2))
    """

    # collect all features that will be used to construct the composite score in a list
    feature_list = [
        item for sublist in composite_score_dict.values() for item in sublist
    ]

    # quality check on composite_score_dict
    if composite_score_dict is None or not bool(composite_score_dict):
        raise AssertionError("Must provide a dictionary with valid feature columns")
    if not all(elem in df.columns for elem in feature_list):
        raise TypeError(
            f" {list(set(feature_list) - set(df.columns))} is not contained in df."
        )

    # def mask_based_on_event(
    #     x: pd.DataFrame, mask_lower_cutoff: Dict[str, list], feature_list: list
    # ):
    #     """Mask values based on threshold if specified"""
    #
    #     if mask_lower_cutoff is not None:
    #         if len(mask_lower_cutoff.keys()) >= 2:
    #             raise AssertionError("mask_lower_cutoff can only handle one key")
    #         else:
    #             (col_event, cutoff_value) = next(iter(mask_lower_cutoff.items()))
    #             x.filter(items=feature_list).mask(df[col_event] <= cutoff_value)
    #     return x

    if mask_lower_cutoff is not None:
        if len(mask_lower_cutoff.keys()) >= 2:
            raise AssertionError("mask_lower_cutoff can only handle one key")
        else:
            mask_lower_cutoff_key = list(mask_lower_cutoff.keys())[0]

            tmp_df = df.filter(items=feature_list).mask(
                df[mask_lower_cutoff_key] <= mask_lower_cutoff[mask_lower_cutoff_key]
            )
            df = pd.merge(
                df.drop(feature_list, axis=1), tmp_df, left_index=True, right_index=True
            )

    # construct features
    composite_score = {
        name: df.filter(items=cluster).apply(method, axis=1)
        for name, cluster in composite_score_dict.items()
    }

    return (
        df
        # .pipe(mask_based_on_event, mask_lower_cutoff, feature_list)
        .assign(**composite_score).pipe(
            lambda x: x.drop(columns=feature_list) if drop_input_features else x
        )
    )
    



    [docs]def construct_composite_score_multicore(
    df: pd.DataFrame,
    composite_score_dicts_list: List[Dict[str, List]],
    drop_input_features: bool = False,
    mask_lower_cutoff: Union[Dict[str, Union[int, float]], None] = None,
    method: Callable = np.mean,
    n_cores: int = 12,
):
    """Construct composite score with flexible type of aggregation.

    Parameters
    ----------
    df: pd.DataFrame
        dataframe containing Neurocast features
    composite_score_dicts_list: list
        a list with dictionaries that specifies the features for each composite score. The keys represent the
        composite score name, whereas the values are lists containing the feature sets used to construct
        the composite score. e.g.
        [{"comp_1": ["feat_1", "feat_2", "feat_3"]},
        {"comp_2": ["feat_4", "feat_5", "feat_6"]},
        {"comp_3": ["feat_7", "feat_8", "feat_9"]},]
    drop_input_features: bool (default: False)
        if True, the features used to construct the composite score are discarded.
    mask_lower_cutoff: Union[dict[str, list], None] (default: None)
        Mask elements based on a cutoff value. The pair {key:value} denotes the column name used for the condition and
        the cutoff value, respectively. For instance, {'event_count': 50} means that within the feature set used to
        construct the composite score, all the rows where event_count <= 50 will be set as np.nan.
    method: Callable (default: np.mean)
        Method passed to pd.DataFrame.apply(axis=1), used to aggregate the feature set contained in the values
        of composite_score_dict. The input can be strings (e.g. 'mean') or functions (e.g. np.mean) Be aware that
        the "multiprocessing" package does not allow "lambda" functions.
    n_cores: int
        An integer specifying the amount of cores to use when calculating the composite score

    Returns
    -------
    pd.Dataframe
        The copy of the original dataframe containing the composite scores.

    Examples
    --------
    >>> import numpy as np
    >>> import pandas as pd
    >>> from neuropy.internal_composite import construct_composite_score_multicore
    >>> import seaborn as sns
    >>> # load dataset
    >>> np.random.seed(seed=42)
    >>> df = (sns
    ...        .load_dataset(name='iris')
    ...        .assign(event_count = np.random.rand(150))
    ...       )
    >>> # construct two composited scores named SEPAL and PETAL using mean aggregation,
    >>> # and mask all value where event_count <= 0.2
    >>> df_out = construct_composite_score_multicore(df=df,
    ...                                   composite_score_dicts_list=[{'SEPAL': ['sepal_length','sepal_width']},
    ...                                                         {'PETAL': ['petal_length','petal_width']}],
    ...                                   mask_lower_cutoff={'event_count': 0.2},
    ...                                   method=np.mean)
    >>> # construct two composite scores named SEPAL and PETAL using np.linalg.norm(ord=2).
    >>> df_out = construct_composite_score_multicore(df=df,
    ...                                   composite_score_dicts_list=[{'SEPAL': ['sepal_length','sepal_width']},
    ...                                                         {'PETAL': ['petal_length','petal_width']}],
    ...                                   method=lambda x: np.linalg.norm(x=x, ord=2))
    """

    # quality check on composite_score_dicts_list
    if (
        len(composite_score_dicts_list) == 0
        or type(composite_score_dicts_list) is not list
        or not all([type(element) is dict for element in composite_score_dicts_list])
    ):
        raise TypeError(
            "Please provide a list of dictionaries as the input for 'composite_score_dicts_list'"
        )

    feature_list = flatten(
        [
            flatten(list(current_dict.values()))
            for current_dict in composite_score_dicts_list
        ]
    )

    if not all(elem in df.columns for elem in feature_list):
        raise TypeError(
            f" {list(set(feature_list) - set(df.columns))} is not contained in df."
        )

    # def mask_based_on_event(
    #     x: pd.DataFrame, mask_lower_cutoff: Dict[str, list], feature_list: list
    # ):
    #     """Mask values based on threshold if specified"""
    #
    #     if mask_lower_cutoff is not None:
    #         if len(mask_lower_cutoff.keys()) >= 2:
    #             raise AssertionError("mask_lower_cutoff can only handle one key")
    #         else:
    #             (col_event, cutoff_value) = next(iter(mask_lower_cutoff.items()))
    #             x.filter(items=feature_list).mask(df[col_event] <= cutoff_value)
    #     return x

    if mask_lower_cutoff is not None:
        if len(mask_lower_cutoff.keys()) >= 2:
            raise AssertionError("mask_lower_cutoff can only handle one key")
        else:
            mask_lower_cutoff_key = list(mask_lower_cutoff.keys())[0]

            tmp_df = df.filter(items=feature_list).mask(
                df[mask_lower_cutoff_key] <= mask_lower_cutoff[mask_lower_cutoff_key]
            )
            df = pd.merge(
                df.drop(feature_list, axis=1), tmp_df, left_index=True, right_index=True
            )

    a_pool = Pool(processes=n_cores)  # specify the amount of cores to use
    start_time = time.time()
    composite_score_list = a_pool.starmap(
        _apply_composite_score,
        [(df, tmp_dict, method) for tmp_dict in composite_score_dicts_list],
    )  # process inputs iterable with pool
    composite_score = pd.concat(
        [pd.DataFrame(tmp_df) for tmp_df in composite_score_list], axis=1
    )
    print("--- %s seconds ---" % (time.time() - start_time))
    a_pool.close()

    return (
        df
        # .pipe(mask_based_on_event, mask_lower_cutoff, feature_list)
        .join(composite_score).pipe(
            lambda x: x.drop(columns=feature_list) if drop_input_features else x
        )
    )
    


def _apply_composite_score(df, cluster_dict, method):
    """worker function used to apply the method per cluster used in the construct_composite_score_multicore function

    Parameters
    ----------
    df: pd.DataFrame
        dataframe containing features present in the cluster_dict
    cluster_dict: dict
        a dictionary with key: string, value: list that specifies the features for the composite score.
         e.g. {"comp_1": ["feat_1", "feat_2", "feat_3"]}
    method: Callable (default: np.mean)
        Method passed to pd.DataFrame.apply(axis=1), used to aggregate the feature set contained in the values
        of composite_score_dict. The input can be strings (e.g. 'mean') or functions (e.g. np.mean). Be aware that
        the "multiprocessing" package does not allow "lambda" functions.

    Returns
    -------
    dict
        The composite score as a dictionary with key: str (the name of the cluster), values: pd.Series (the scores
        corresponding to the calculated cluster; the length of the pd.Series will match the length of the inputted
        pd.DataFrame).

    Examples
    --------
    >>> import numpy as np
    >>> import pandas as pd
    >>> from neuropy.internal_composite import construct_composite_score_multicore
    >>> import seaborn as sns
    >>> # load dataset
    >>> np.random.seed(seed=42)
    >>> df = (sns
    ...        .load_dataset(name='iris')
    ...        .assign(event_count = np.random.rand(150))
    ...       )
    >>> dict_out = _apply_composite_score(df=df,
    ...                              cluster_dict={'SEPAL': ['sepal_length','sepal_width']},
    ...                              method=np.mean)

    """

    composite_score = {}
    # start_time = time.time()
    (name, cluster) = flatten(list(cluster_dict.items()))
    composite_score[name] = df.filter(items=cluster).apply(method, axis=1)
    # print("--- %s seconds ---" % (time.time() - start_time))
    return composite_score
